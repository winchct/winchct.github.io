function uuid_to_username(uuid) { if (is_string($uuid)) { uuid = str_replace('-', '', uuid); if (strlen(uuid) === 32 and ctype_alnum(uuid)) { json = file_get_contents('https://api.mojang.com/user/profiles/' . uuid . '/names'); if (!empty(json)) { data = json_decode(json, true); if (!empty($data) and is_array(data)) { $last = array_pop(data); if (is_array(last) and isset(last['name'])) { return last['name']; } } } } } return false; }
